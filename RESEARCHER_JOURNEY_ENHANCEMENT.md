## Priority 3: Enhanced Timeline with Dependencies

**Problem:** Basic milestones only. No task breakdown, no dependencies, no per-person allocation.

### Enhanced Milestone Model

```typescript
// Current (basic)
interface Milestone {
  name: string;
  startDate: string;
  endDate: string;
  status: "planned" | "in-progress" | "completed" | "delayed";
  description: string;
}

// Enhanced (proposed)
interface Milestone {
  id: string;
  name: string;
  startDate: string;
  endDate: string;
  status: "planned" | "in-progress" | "completed" | "delayed";
  description: string;

  // NEW: Dependencies
  dependsOn: string[];           // IDs of prerequisite milestones

  // NEW: Tasks within milestone
  tasks: Task[];

  // NEW: Resource allocation
  assignedTeamMembers: {
    memberId: number;
    hoursAllocated: number;
    role: string;
  }[];

  // NEW: Deliverables
  expectedOutputs: string[];     // IDs of deliverables

  // NEW: Budget link
  budgetCategory?: string;
  estimatedCost?: number;
}

interface Task {
  id: string;
  name: string;
  description: string;
  startDate: string;
  endDate: string;
  status: "pending" | "in-progress" | "completed";
  assignedTo: number;            // Team member ID
  estimatedHours: number;
  actualHours?: number;
}
```

```sql
-- Team allocation enhancements
ALTER TABLE project_collaborators ADD COLUMN hours_per_week DECIMAL(4,1);
ALTER TABLE project_collaborators ADD COLUMN hourly_rate DECIMAL(10,2);
ALTER TABLE project_collaborators ADD COLUMN responsibilities TEXT;
ALTER TABLE project_collaborators ADD COLUMN start_date DATE;
ALTER TABLE project_collaborators ADD COLUMN end_date DATE;

-- New outputs table
CREATE TABLE project_outputs (
  id INT AUTO_INCREMENT PRIMARY KEY,
  project_id INT NOT NULL,
  type VARCHAR(50) NOT NULL,
  title VARCHAR(255) NOT NULL,
  description TEXT,
  target_date DATE,
  status VARCHAR(20) DEFAULT 'planned',
  linked_milestone_id VARCHAR(50),
  metadata JSON,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (project_id) REFERENCES projects(id)
);

-- Milestone tasks table
CREATE TABLE milestone_tasks (
  id INT AUTO_INCREMENT PRIMARY KEY,
  project_id INT NOT NULL,
  milestone_id VARCHAR(50) NOT NULL,
  name VARCHAR(255) NOT NULL,
  description TEXT,
  assigned_to INT,
  estimated_hours DECIMAL(6,1),
  actual_hours DECIMAL(6,1),
  start_date DATE,
  end_date DATE,
  status VARCHAR(20) DEFAULT 'pending',
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (project_id) REFERENCES projects(id),
  FOREIGN KEY (assigned_to) REFERENCES users(id)
);

-- Milestone dependencies table
CREATE TABLE milestone_dependencies (
  id INT AUTO_INCREMENT PRIMARY KEY,
  project_id INT NOT NULL,
  milestone_id VARCHAR(50) NOT NULL,
  depends_on_milestone_id VARCHAR(50) NOT NULL,
  dependency_type VARCHAR(20) DEFAULT 'finish_to_start',
  lag_days INT DEFAULT 0,
  FOREIGN KEY (project_id) REFERENCES projects(id)
);
```
