## New Database Schema

```sql
-- =====================================================
-- TASK MANAGEMENT
-- =====================================================

CREATE TABLE tasks (
  id INT AUTO_INCREMENT PRIMARY KEY,
  project_id INT NOT NULL,
  milestone_id VARCHAR(50),
  parent_task_id INT,                    -- For subtasks
  title VARCHAR(255) NOT NULL,
  description TEXT,
  status ENUM('backlog', 'todo', 'in_progress', 'review', 'done') DEFAULT 'backlog',
  priority ENUM('low', 'medium', 'high', 'urgent') DEFAULT 'medium',
  assigned_to INT,
  estimated_hours DECIMAL(6,1),
  actual_hours DECIMAL(6,1) DEFAULT 0,
  due_date DATE,
  completed_at TIMESTAMP,
  created_by INT NOT NULL,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (project_id) REFERENCES projects(id),
  FOREIGN KEY (assigned_to) REFERENCES users(id),
  FOREIGN KEY (parent_task_id) REFERENCES tasks(id)
);

CREATE TABLE task_comments (
  id INT AUTO_INCREMENT PRIMARY KEY,
  task_id INT NOT NULL,
  user_id INT NOT NULL,
  content TEXT NOT NULL,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (task_id) REFERENCES tasks(id),
  FOREIGN KEY (user_id) REFERENCES users(id)
);

CREATE TABLE task_attachments (
  id INT AUTO_INCREMENT PRIMARY KEY,
  task_id INT NOT NULL,
  file_name VARCHAR(255) NOT NULL,
  file_path VARCHAR(500) NOT NULL,
  file_size INT,
  uploaded_by INT NOT NULL,
  uploaded_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (task_id) REFERENCES tasks(id),
  FOREIGN KEY (uploaded_by) REFERENCES users(id)
);

CREATE TABLE time_entries (
  id INT AUTO_INCREMENT PRIMARY KEY,
  task_id INT NOT NULL,
  user_id INT NOT NULL,
  hours DECIMAL(4,2) NOT NULL,
  description TEXT,
  entry_date DATE NOT NULL,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (task_id) REFERENCES tasks(id),
  FOREIGN KEY (user_id) REFERENCES users(id)
);

-- =====================================================
-- PARTICIPANT / DATA COLLECTION
-- =====================================================

CREATE TABLE participants (
  id INT AUTO_INCREMENT PRIMARY KEY,
  project_id INT NOT NULL,
  participant_id VARCHAR(50) NOT NULL,   -- Study ID (e.g., P-001)
  status ENUM('screening', 'enrolled', 'active', 'on_hold', 'completed', 'withdrawn') DEFAULT 'screening',
  enrollment_date DATE,
  withdrawal_date DATE,
  withdrawal_reason TEXT,
  site_id INT,
  assigned_coordinator INT,
  demographics JSON,                      -- Age, gender, etc.
  contact_info JSON,                      -- Encrypted contact details
  notes TEXT,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (project_id) REFERENCES projects(id),
  FOREIGN KEY (assigned_coordinator) REFERENCES users(id),
  UNIQUE KEY (project_id, participant_id)
);

CREATE TABLE study_visits (
  id INT AUTO_INCREMENT PRIMARY KEY,
  project_id INT NOT NULL,
  participant_id INT NOT NULL,
  visit_number INT NOT NULL,
  visit_name VARCHAR(100),
  scheduled_date DATE,
  actual_date DATE,
  window_start DATE,
  window_end DATE,
  status ENUM('scheduled', 'completed', 'missed', 'rescheduled') DEFAULT 'scheduled',
  completed_by INT,
  notes TEXT,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (project_id) REFERENCES projects(id),
  FOREIGN KEY (participant_id) REFERENCES participants(id),
  FOREIGN KEY (completed_by) REFERENCES users(id)
);

CREATE TABLE data_forms (
  id INT AUTO_INCREMENT PRIMARY KEY,
  project_id INT NOT NULL,
  form_name VARCHAR(100) NOT NULL,
  form_version VARCHAR(20),
  form_schema JSON NOT NULL,              -- Field definitions
  is_active BOOLEAN DEFAULT TRUE,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (project_id) REFERENCES projects(id)
);

CREATE TABLE form_entries (
  id INT AUTO_INCREMENT PRIMARY KEY,
  form_id INT NOT NULL,
  visit_id INT NOT NULL,
  participant_id INT NOT NULL,
  data JSON NOT NULL,                     -- Actual form data
  status ENUM('draft', 'complete', 'verified', 'locked') DEFAULT 'draft',
  entered_by INT NOT NULL,
  verified_by INT,
  entered_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  verified_at TIMESTAMP,
  FOREIGN KEY (form_id) REFERENCES data_forms(id),
  FOREIGN KEY (visit_id) REFERENCES study_visits(id),
  FOREIGN KEY (participant_id) REFERENCES participants(id),
  FOREIGN KEY (entered_by) REFERENCES users(id),
  FOREIGN KEY (verified_by) REFERENCES users(id)
);

CREATE TABLE data_queries (
  id INT AUTO_INCREMENT PRIMARY KEY,
  form_entry_id INT NOT NULL,
  field_name VARCHAR(100) NOT NULL,
  query_text TEXT NOT NULL,
  status ENUM('open', 'answered', 'closed') DEFAULT 'open',
  raised_by INT NOT NULL,
  answered_by INT,
  answer_text TEXT,
  raised_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  answered_at TIMESTAMP,
  closed_at TIMESTAMP,
  FOREIGN KEY (form_entry_id) REFERENCES form_entries(id),
  FOREIGN KEY (raised_by) REFERENCES users(id),
  FOREIGN KEY (answered_by) REFERENCES users(id)
);

-- =====================================================
-- DOCUMENT MANAGEMENT
-- =====================================================

CREATE TABLE documents (
  id INT AUTO_INCREMENT PRIMARY KEY,
  project_id INT NOT NULL,
  folder_path VARCHAR(500) NOT NULL,
  file_name VARCHAR(255) NOT NULL,
  file_path VARCHAR(500) NOT NULL,
  file_size INT,
  mime_type VARCHAR(100),
  version VARCHAR(20) DEFAULT '1.0',
  status ENUM('draft', 'under_review', 'approved', 'superseded', 'archived') DEFAULT 'draft',
  uploaded_by INT NOT NULL,
  approved_by INT,
  approved_at TIMESTAMP,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (project_id) REFERENCES projects(id),
  FOREIGN KEY (uploaded_by) REFERENCES users(id),
  FOREIGN KEY (approved_by) REFERENCES users(id)
);

CREATE TABLE document_versions (
  id INT AUTO_INCREMENT PRIMARY KEY,
  document_id INT NOT NULL,
  version VARCHAR(20) NOT NULL,
  file_path VARCHAR(500) NOT NULL,
  change_summary TEXT,
  uploaded_by INT NOT NULL,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (document_id) REFERENCES documents(id),
  FOREIGN KEY (uploaded_by) REFERENCES users(id)
);

-- =====================================================
-- COMPLIANCE & SAFETY
-- =====================================================

CREATE TABLE protocol_deviations (
  id INT AUTO_INCREMENT PRIMARY KEY,
  project_id INT NOT NULL,
  participant_id INT,
  deviation_date DATE NOT NULL,
  description TEXT NOT NULL,
  severity ENUM('minor', 'moderate', 'major') NOT NULL,
  category VARCHAR(100),
  root_cause TEXT,
  corrective_action TEXT,
  status ENUM('identified', 'under_review', 'reported', 'resolved') DEFAULT 'identified',
  reported_by INT NOT NULL,
  reviewed_by INT,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (project_id) REFERENCES projects(id),
  FOREIGN KEY (participant_id) REFERENCES participants(id),
  FOREIGN KEY (reported_by) REFERENCES users(id)
);

CREATE TABLE adverse_events (
  id INT AUTO_INCREMENT PRIMARY KEY,
  project_id INT NOT NULL,
  participant_id INT NOT NULL,
  event_date DATE NOT NULL,
  description TEXT NOT NULL,
  severity ENUM('mild', 'moderate', 'severe', 'life_threatening', 'fatal') NOT NULL,
  relationship ENUM('unrelated', 'unlikely', 'possible', 'probable', 'definite'),
  is_serious BOOLEAN DEFAULT FALSE,
  outcome ENUM('ongoing', 'resolved', 'resolved_with_sequelae', 'fatal') DEFAULT 'ongoing',
  outcome_date DATE,
  reported_by INT NOT NULL,
  reviewed_by INT,
  status ENUM('reported', 'under_review', 'closed') DEFAULT 'reported',
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (project_id) REFERENCES projects(id),
  FOREIGN KEY (participant_id) REFERENCES participants(id),
  FOREIGN KEY (reported_by) REFERENCES users(id)
);

-- =====================================================
-- FINANCIAL TRACKING
-- =====================================================

CREATE TABLE expenses (
  id INT AUTO_INCREMENT PRIMARY KEY,
  project_id INT NOT NULL,
  category VARCHAR(50) NOT NULL,
  description TEXT NOT NULL,
  amount DECIMAL(10,2) NOT NULL,
  currency VARCHAR(3) DEFAULT 'USD',
  expense_date DATE NOT NULL,
  vendor VARCHAR(255),
  receipt_path VARCHAR(500),
  status ENUM('pending', 'approved', 'rejected', 'reimbursed') DEFAULT 'pending',
  submitted_by INT NOT NULL,
  approved_by INT,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (project_id) REFERENCES projects(id),
  FOREIGN KEY (submitted_by) REFERENCES users(id),
  FOREIGN KEY (approved_by) REFERENCES users(id)
);

-- =====================================================
-- COMMUNICATION
-- =====================================================

CREATE TABLE channels (
  id INT AUTO_INCREMENT PRIMARY KEY,
  project_id INT NOT NULL,
  name VARCHAR(50) NOT NULL,
  description TEXT,
  is_private BOOLEAN DEFAULT FALSE,
  created_by INT NOT NULL,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (project_id) REFERENCES projects(id),
  FOREIGN KEY (created_by) REFERENCES users(id)
);

CREATE TABLE messages (
  id INT AUTO_INCREMENT PRIMARY KEY,
  channel_id INT,
  sender_id INT NOT NULL,
  recipient_id INT,                       -- For direct messages
  content TEXT NOT NULL,
  is_read BOOLEAN DEFAULT FALSE,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (channel_id) REFERENCES channels(id),
  FOREIGN KEY (sender_id) REFERENCES users(id),
  FOREIGN KEY (recipient_id) REFERENCES users(id)
);

CREATE TABLE notifications (
  id INT AUTO_INCREMENT PRIMARY KEY,
  user_id INT NOT NULL,
  project_id INT,
  type VARCHAR(50) NOT NULL,
  title VARCHAR(255) NOT NULL,
  message TEXT,
  link VARCHAR(500),
  is_read BOOLEAN DEFAULT FALSE,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (user_id) REFERENCES users(id),
  FOREIGN KEY (project_id) REFERENCES projects(id)
);

-- =====================================================
-- AUDIT TRAIL
-- =====================================================

CREATE TABLE audit_log (
  id INT AUTO_INCREMENT PRIMARY KEY,
  project_id INT,
  user_id INT NOT NULL,
  action VARCHAR(50) NOT NULL,
  entity_type VARCHAR(50) NOT NULL,
  entity_id INT,
  old_values JSON,
  new_values JSON,
  ip_address VARCHAR(45),
  user_agent TEXT,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (project_id) REFERENCES projects(id),
  FOREIGN KEY (user_id) REFERENCES users(id),
  INDEX idx_audit_project (project_id),
  INDEX idx_audit_user (user_id),
  INDEX idx_audit_entity (entity_type, entity_id)
);
```
